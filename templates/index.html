def generate_signal(price): if price is None: return { "price": 0, "signal": "WAIT", "trend": "NO DATA", "entry": "NO TRADE", "confidence": 0, "risk": "HIGH", "sl": 0, "target": 0 } support = round(price - 80, 2) resistance = round(price + 80, 2) zone = (resistance - support) * 0.2 # 🔥 breakout zone # 🔥 STRONG BUY (REAL BREAKOUT) if price > resistance - zone: signal = "STRONG BUY 🔥" trend = "BULLISH 📈" entry = "ENTER NOW 🟢" confidence = 90 # 🔥 STRONG SELL (REAL BREAKDOWN) elif price < support + zone: signal = "STRONG SELL 🔻" trend = "BEARISH 📉" entry = "ENTER NOW 🔴" confidence = 90 # ⚖️ SIDEWAYS ZONE elif support + zone < price < resistance - zone: signal = "HOLD ⚠️" trend = "SIDEWAYS" entry = "NO TRADE ❌" confidence = 50 # fallback else: signal = "WAIT" trend = "NEUTRAL" entry = "WAIT ⏳" confidence = 60 risk = "LOW" if confidence >= 85 else "MEDIUM" if confidence >= 70 else "HIGH" return { "price": round(price,2), "signal": signal, "trend": trend, "entry": entry, "confidence": confidence, "risk": risk, "sl": support, "target": resistance }
